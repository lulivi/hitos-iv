module hug_hitos_iv.

import hug
import hitos_iv

json_path = '../data/hitos.json'
hitos_iv_json = hitos_iv.read_json(json_path)
#

=== Get / === Retunr web status.

@hug.get('/')
def index():
# 
    return {'status': 'OK'}
#

=== Get /all === Return all milestones.

@hug.get('/all')
def get_all():
# 
    return hitos_iv.read_json(json_path)
#

=== Get /number === Return all milestones.

@hug.get('/number')
def milestone_number():
# 
    return hitos_iv.milestone_number(hitos_iv_json)
#

=== Get /get/{id} === Return one milestone.

@hug.get('/get/{id}')
def get_one(id: int):
# 
    return hitos_iv.get_milestone(hitos_iv_json, id)
